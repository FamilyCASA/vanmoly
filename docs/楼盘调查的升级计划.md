### 4.2 通用 AI 工具封装 `ai_utils.py`

```python
# ai_utils.py
import requests
from ai_config import AI_API_KEY, AI_API_URL


def ai_analysis(prompt: str, content: str) -> str:
    """
    通用 AI 分析接口
    :param prompt: 给 AI 的指令（如：总结楼盘配套、提取客户信息）
    :param content: 原始文本内容
    :return: AI 整理好的结果
    """
    messages = [
        {"role": "system", "content": "你是楼盘数据整理专员，输出简洁结构化文字，适合直接填入表单"},
        {"role": "user", "content": f"{prompt}\n原始内容：{content}"}
    ]

    payload = {
        "model": "general",
        "messages": messages,
        "temperature": 0.3
    }

    headers = {
        "Authorization": f"Bearer {AI_API_KEY}",
        "Content-Type": "application/json"
    }

    try:
        res = requests.post(AI_API_URL, json=payload, headers=headers, timeout=30)
        res_json = res.json()
        # 兼容主流大模型返回格式
        return res_json["choices"][0]["message"]["content"].strip()
    except Exception as e:
        return f"AI解析失败：{str(e)}"


def ai_summary_building(raw_text: str) -> str:
    """楼盘信息总结专用"""
    prompt = "帮我整理精简为楼盘表单可用的结构化文案，分点清晰，不要多余废话"
    return ai_analysis(prompt, raw_text)


def ai_extract_customer(ocr_text: str) -> str:
    """OCR 文字提取客户信息：姓名、电话、房号"""
    prompt = "从以下文字中提取所有业主信息，只输出：姓名|手机号|房号，一行一条，无多余文字"
    return ai_analysis(prompt, ocr_text)
```

### 4.3 OCR 识别模块 `ocr_api.py`

> 预留接口，填入密钥即可使用

```python
# ocr_api.py
import requests
from ai_config import OCR_API_KEY, OCR_API_SECRET, OCR_API_URL


def ocr_image_recognize(image_base64: str) -> str:
    """传入图片 base64，返回识别纯文本"""
    payload = {
        "image": image_base64,
        "key": OCR_API_KEY,
        "secret": OCR_API_SECRET
    }
    try:
        res = requests.post(OCR_API_URL, json=payload, timeout=20)
        data = res.json()
        return data.get("text", "")
    except:
        return "OCR识别失败"
```

### 4.6 后端主程序 `main.py`

```python
# main.py — 启动入口
from fastapi import FastAPI, UploadFile, Form
from fastapi.middleware.cors import CORSMiddleware
import base64

from crawl_api import crawl_building_by_name
from ai_utils import ai_extract_customer
from ocr_api import ocr_image_recognize
from excel_import import read_customer_excel

app = FastAPI(title="楼盘智能表单AI接口")

# 跨域允许前端访问
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)


# 接口1：楼盘名称抓取 + AI 总结
@app.get("/api/crawl_build")
def api_crawl_build(build_name: str):
    return crawl_building_by_name(build_name)


# 接口2：上传图片 OCR + AI 提取客户
@app.post("/api/ocr_customer")
async def api_ocr_customer(file: UploadFile):
    img_bytes = await file.read()
    img_base64 = base64.b64encode(img_bytes).decode()
    ocr_text = ocr_image_recognize(img_base64)
    ai_text = ai_extract_customer(ocr_text)
    return {"code": 200, "ocr_text": ocr_text, "ai_text": ai_text}


# 接口3：Excel 批量导入线索
@app.post("/api/import_excel")
async def api_import_excel(file: UploadFile):
    save_path = "temp.xlsx"
    with open(save_path, "wb") as f:
        f.write(await file.read())
    data = read_customer_excel(save_path)
    return {"code": 200, "data": data}


if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="127.0.0.1", port=8000)
```

## 五、使用步骤

1. **创建目录**：按上方目录结构建好 `loupan_form/` 文件夹
2. **安装依赖**：`pip install fastapi uvicorn requests pandas openpyxl`
3. **填写密钥**：打开 `ai_config.py`，填入你的 AI 大模型 Key、接口地址、OCR 密钥
4. **启动后端**：`python main.py`（默认监听 `127.0.0.1:8000`）
5. **打开前端**：双击 `index.html` 即可使用
   - 输入楼盘名 → 一键抓取 → AI 自动总结填表，可手动二次修改
   - 上传业主签到照片/扫描件 → 自动 OCR + AI 提取姓名电话房号
   - 上传 Excel → 批量导入客户线索

## 九、注意事项

1. **密钥安全**：`ai_config.py` 中的 API 密钥不要提交到公开仓库，建议加入 `.gitignore`
2. **OCR 接口选型**：百度 OCR 通用文字识别每天有免费额度，适合初期使用
3. **爬虫合规**：真实爬虫需遵守目标网站 robots.txt，控制请求频率
4. **AI 输出校验**：AI 总结结果建议人工复核后再正式使用
5. **图片隐私**：业主扫描件含个人信息，传输和存储需注意数据保护
