})

// 莽聜鹿氓聡禄忙聲掳猫陆麓盲禄禄忙聞聫盲陆聧莽陆庐 芒聠聮 茅聙聣盲赂颅猫炉楼莽聜鹿 脗卤10盲赂聡
const clickAxis = (event) => {
  const track = event.currentTarget
  const rect = track.getBoundingClientRect()
  const pct = Math.max(0, Math.min(1, (event.clientX - rect.left) / rect.width))
  const min = budgetStats.value.min_wan
  const max = budgetStats.value.max_wan
  const midWan = min + pct * (max - min)
  const minWan = Math.max(0, midWan - 5)
  const maxWan = midWan + 5
  filters.price_min = Math.floor(minWan * 10000)
  filters.price_max = Math.ceil(maxWan * 10000)
  budgetRange.value = true
  selectedQuickRange.value = ''
  // 忙禄聭氓聺聴莽搂禄氓聤篓氓聢掳莽聜鹿氓聡禄盲陆聧莽陆庐
  dragPct.value = pct * 100
  // 莽禄聼猫庐隆猫炉楼氓聦潞茅聴麓忙隆聢盲戮聥忙聲掳茂录聢氓聣聧莽芦炉盲录掳莽庐聴茂录聣
  budgetTip.value = estimateCount(midWan)
  fetchCases()
}

// 盲录掳莽庐聴氓聦潞茅聴麓氓聠聟忙隆聢盲戮聥忙聲掳茂录聢氓聼潞盲潞聨 buckets 忙聲掳忙聧庐莽虏聴盲录掳茂录聣
const estimateCount = (midWan) => {
  const buckets = budgetStats.value?.buckets || []
  const low = (midWan - 5) * 10000
  const high = (midWan + 5) * 10000
  return buckets.reduce((sum, b) => {
    const bm = b.mid || 0
    return sum + (bm >= low && bm <= high ? b.count : 0)
  }, 0)
}

// 氓驴芦忙聧路氓聦潞茅聴麓忙聦聣茅聮庐
const applyQuickRange = (range) => {
  selectedQuickRange.value = range.label
  filters.price_min = range.min
  filters.price_max = range.max
  budgetRange.value = true
  budgetTip.value = estimateCount((range.min + (range.max || range.min * 2)) / 20000)
  fetchCases()
}

// ===== 忙禄聭氓聺聴忙聥聳忙聥陆 =====
const startDrag = (e) => {
  isDragging.value = true
  dragPct.value = null
  document.addEventListener('mousemove', onDrag)
  document.addEventListener('mouseup', endDrag)
  e.preventDefault()
}

const onDrag = (e) => {
  if (!isDragging.value || !budgetStats.value) return
  const track = document.querySelector('.budget-axis-track')
  if (!track) return
  const rect = track.getBoundingClientRect()
  const pct = Math.max(0, Math.min(100, ((e.clientX - rect.left) / rect.width) * 100))
  dragPct.value = pct
  // 氓庐聻忙聴露茅垄聞猫搂聢茂录職猫庐隆莽庐聴氓炉鹿氓潞聰盲赂聡忙聲掳氓鹿露忙聸麓忙聳掳忙聫聬莽陇潞
  const min = budgetStats.value.min_wan
  const max = budgetStats.value.max_wan
  const midWan = min + (pct / 100) * (max - min)
  budgetTip.value = estimateCount(midWan)
}

const endDrag = () => {
  if (!isDragging.value) {
    document.removeEventListener('mousemove', onDrag)
    document.removeEventListener('mouseup', endDrag)
    return
  }
  if (dragPct.value !== null && budgetStats.value) {
    const min = budgetStats.value.min_wan
    const max = budgetStats.value.max_wan
    const midWan = min + (dragPct.value / 100) * (max - min)
    // 猫庐戮莽陆庐 脗卤10盲赂聡 氓聦潞茅聴麓
    const minWan = Math.max(0, midWan - 10)
    const maxWan = midWan + 10
    filters.price_min = Math.floor(minWan * 10000)
    filters.price_max = Math.ceil(maxWan * 10000)
    budgetRange.value = true
    selectedQuickRange.value = ''
    fetchCases()
  }
  isDragging.value = false
  dragPct.value = null
  document.removeEventListener('mousemove', onDrag)
  document.removeEventListener('mouseup', endDrag)
}

// 忙禄聭氓聺聴忙聣聥忙聼聞莽聜鹿氓聡禄茂录聢盲禄聟氓聹篓茅聺聻忙聥聳忙聥陆忙聴露猫搂娄氓聫聭茂录聣
const handleClick = () => {
  if (isDragging.value) return
  budgetRange.value = !budgetRange.value
  if (!budgetRange.value) {
    filters.price_min = null
    filters.price_max = null
    budgetTip.value = 0
    fetchCases()
  }
}
